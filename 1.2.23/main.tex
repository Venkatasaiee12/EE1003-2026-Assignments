\section{Problem}

\begin{frame}
\frametitle{Problem}
The minimum frequency at which a signal comprising of 30 Hz, 50 Hz and 70 Hz should be sampled to avoid aliasing is \dots Hz.
\end{frame}
%\subsection{Literature}
\section{Solution}

\subsection{Sampling}
\setcounter{section}{1}
\begin{frame}
\frametitle{Sampling}
 Sampling is the process of converting a continuous-time analog signal into a discrete-time digital signal. This is done by measuring the amplitude of the signal at uniform intervals, defined by the Sampling Frequency.
\end{frame}
\subsection{The Nyquist-Shannon Sampling Theorem}
\begin{frame}
\frametitle{The Nyquist-Shannon Sampling Theorem}
To ensure a signal can be perfectly reconstructed without losing information, it must be sampled at a rate at least twice the highest frequency component present in the signal. This minimum required rate is known as the Nyquist Rate.
\begin{align}
f_s\geq 2 \times f_{max}
  \end{align}
\end{frame}
\subsection{Aliasing}
\begin{frame}
\frametitle{Aliasing}
 If the sampling frequency is lower than the Nyquist Rate, aliasing occurs. This is a phenomenon where high-frequency components appear as lower-frequency "aliases" in the reconstructed data. These fake lower frequencies distort the original signal and cannot be removed once sampled.

 \end{frame}
 \subsection{Calculation}
\begin{frame}
\frametitle{Calculation}
Given frequencies = $\cbrak{30,50,70}\ \text{Hz}$ 
\begin{align}
    f_{max}& = 70\ \text{Hz}
\end{align}
From Nyquist-Shannon Theorem,
\begin{align}
   f_s =  2 \times f_{max}  
\end{align}
\begin{align}
    f_s =  2 \times 70\ \text{Hz} \implies f_{s} =  140\ \text{Hz}
\end{align}
 \end{frame}
\begin{frame}
\subsection{Analysis on Undersampling}
\frametitle{Analysis on Undersampling}
If this signal were sampled at only 100 Hz (which is $\leq$ 140 Hz):
\begin{enumerate}
    \item The 30 Hz and 50 Hz components would be captured correctly.
    \item The 70 Hz component would alias. It would appear as a 30 Hz signal  interfering with the original 30 Hz component and causing irreversible distortion.
\end{enumerate}
To find the alias frequency $f_a$ for any signal $f_{in}$ sampled at a rate $f_{s}$ :
\begin{align}
    f_a &= |f_s-f_{in}| \\
     f_a &= |100-70| =30\ \text{Hz}
    \end{align}
In this case, this 30 Hz might get overlapped with the original 30 Hz signal. This make impossible to seperate the aliased one and the original one later
\end{frame}

import numpy as np
import matplotlib.pyplot as plt
from scipy.fft import rfft, rfftfreq

freqs = [30, 50, 70]

def sig(t):
    return sum(np.cos(2*np.pi*f*t) for f in freqs)

duration = 0.5
t_cont = np.linspace(0, duration, 2000, endpoint=False)
x_cont = sig(t_cont)

fig, ax = plt.subplots(2, 2, figsize=(12,6))

for i, fs in enumerate([150, 100]):
    t = np.arange(0, duration, 1/fs)
    x = sig(t)
