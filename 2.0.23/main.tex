\section{Problem}

\begin{frame}
\frametitle{Problem}
The maximum value of the function $2x + 3y + 4z$ on the ellipsoid $2x^2 + 3y^2 + 4z^2 = 1$ is:
\end{frame}
%\subsection{Literature}
\section{Solution}

\subsection{Representation}
\setcounter{section}{1}
\begin{frame}
\frametitle{Representation}
The equation of the conic is given by
\begin{align}
     \vec{x}^\top\vec{V}\vec{x} + 2\vec{u}^\top\vec{x} + f = 0
\end{align}
On comparing the given ellipsoid $2x^2 + 3y^2 + 4z^2 = 1$ , we get

\medskip
\begin{align}
\vec{V} =  \myvec{ 2 & 0 & 0 \\ 0 & 3 & 0 \\ 0 & 0 & 4 },  \vec{u} = \myvec{ 0 \\ 0 \\ 0 }\  \text{and}\ f = -1 
\end{align}
The function $2x + 3y + 4z$ = c is a plane with:
\begin{align}
    \vec{n}^\top\vec{x} = c 
\end{align}
where
\begin{align}
\vec{n} = \myvec{ 2 \\ 3 \\ 4 }
\end{align}
\end{frame}
\subsection{Tangency Condition}
\begin{frame}
\frametitle{Tangency Condition}
 If $\vec{V}^{-1}$ exists, given the normal vector $\vec{n}$, the tangent points of contact to conic are given by
\begin{align}
     \vec{q} = &\vec{V}^{-1}(\kappa_{i} \vec{n} - \vec{u}) 
\end{align}
where 
\begin{align}
    \kappa_{i} = 
    \pm\sqrt{\frac{f_0}{\vec{n}^\top\vec{V}^{-1}\vec{n}}}\ \text{and}\ f_0 = \vec{u}^\top\vec{V}^{-1}\vec{u} - f
\end{align}
\end{frame}
\subsection{Calculation}
\begin{frame}
\frametitle{Calculation}
By substituting $\vec{q}$ in (1.3)
\begin{align}
 c = \vec{n}^\top \vec{V}^{-1}(\kappa \vec{n} - \vec{u}) &= \pm \frac{f_o}{\sqrt{\vec{n}^\top\vec{V}^{-1}\vec{n}}}\brak{\vec{n}^\top\vec{V}^{-1}\vec{n}} - \vec{n}^\top\vec{V}^{-1}\vec{u}\\
 &=\pm\sqrt{f_0 (\vec{n}^\top\vec{V}^{-1}\vec{n})} - \vec{n}^\top\vec{V}^{-1}\vec{u} \\
 &=\pm\sqrt{\brak{\vec{u}^\top\vec{V}^{-1}\vec{u} - f}(\vec{n}^\top\vec{V}^{-1}\vec{n})} - \vec{n}^\top\vec{V}^{-1}\vec{u}
 \end{align}
 Given 
 \begin{align}
     \vec{V} =  \myvec{ 2 & 0 & 0 \\ 0 & 3 & 0 \\ 0 & 0 & 4 } \implies 
     \vec{V}^{-1} =  \myvec{ \frac{1}{2} & 0 & 0 \\ 0 & \frac{1}{3} & 0 \\ 0 & 0 & \frac{1}{4} }
 \end{align}
 Substituting given values from (1.2) 

 As $\vec{u}=\vec{0}$,
 \begin{align}
     \vec{u}^\top\vec{V}^{-1}\vec{u} = 0\ \text{and}\  \vec{n}^\top\vec{V}^{-1}\vec{u} = 0
     \end{align}
\end{frame}

\begin{frame}
\frametitle{Calculation}
 \begin{align}
     \vec{n}^\top \mathbf{V}^{-1} \vec{n} &= \myvec{2 & 3 & 4} \myvec{ \frac{1}{2} & 0 & 0 \\ 0 & \frac{1}{3} & 0 \\ 0 & 0 & \frac{1}{4} } \myvec{2 \\ 3 \\ 4} \\
     &= \myvec{1 & 1 &1}\myvec{2\\3\\4} \\
     &=  (1 \cdot 2) + (1 \cdot 3) + (1 \cdot 4) = 9
 \end{align}
 Substituting these values in (1.9), we get
 \begin{align}
c = \pm \sqrt{9}-0\\
c = \pm \sqrt{9} = \pm 3
 \end{align}
 Hence the maximum value of c is 3.
 \end{frame}

\begin{frame}
\frametitle{Finding the point}
From (1.5),
 \begin{align}
    \vec{q} = &\vec{V}^{-1}(\kappa_{i} \vec{n} - \vec{u}) 
 \end{align}
 \begin{align}
 f_{o} = \vec{u}^\top\vec{V}^{-1}\vec{u} - f = 0 - (-1) = 1 \\
     \kappa_{i} = 
    \pm\sqrt{\frac{f_0}{\vec{n}^\top\vec{V}^{-1}\vec{n}}}=\pm \sqrt{\frac{1}{9}}=\pm\frac{1}{3} \\
    \vec{q} = \myvec{ \frac{1}{2} & 0 & 0 \\ 0 & \frac{1}{3} & 0 \\ 0 & 0 & \frac{1}{4} }\brak{\pm\frac{1}{3} \myvec{2 \\ 3\\ 4 } - \vec{0}}  \\
    = \pm\frac{1}{3}\myvec{ \frac{1}{2} & 0 & 0 \\ 0 & \frac{1}{3} & 0 \\ 0 & 0 & \frac{1}{4} } \myvec{2 \\ 3\\ 4 } = \myvec{ \pm\frac{1}{3} \\ \pm\frac{1}{3} \\ \pm\frac{1}{3}   }
 \end{align}
 
 \end{frame}

import cvxpy as cp
import numpy as np
import matplotlib.pyplot as plt
from matplotlib.lines import Line2D

# Define the variables
x_var = cp.Variable(3)

# Define Matrix V and vector n
V_mat = np.diag([2, 3, 4])
n_vec = np.array([2, 3, 4])

# Maximize n^T x subject to x^T V x <= 1
objective = cp.Maximize(n_vec @ x_var)
constraints = [cp.quad_form(x_var, V_mat) <= 1]
prob = cp.Problem(objective, constraints)

\end{lstlisting}
\end{frame}
 
\begin{frame}[fragile]
\frametitle{Python Code for Plotting}
\begin{lstlisting}[language=Python]
# Solve the problem
prob.solve()

# Results
c_max = prob.value
q_max = x_var.value
q_min = -q_max  # Symmetry of the ellipsoid

print(f"Max value found: {c_max:.4f}")
print(f"Point of maximization: {q_max}")

# 2. Visualization setup
fig = plt.figure(figsize=(12, 8))
ax = fig.add_subplot(111, projection='3d')

# Generate Ellipsoid Data
u = np.linspace(0, 2 * np.pi, 100)
v = np.linspace(0, np.pi, 100)

\end{lstlisting}
\end{frame}

rx, ry, rz = 1/np.sqrt(2), 1/np.sqrt(3), 1/np.sqrt(4)
x_ell = rx * np.outer(np.cos(u), np.sin(v))
y_ell = ry * np.outer(np.sin(u), np.sin(v))
z_ell = rz * np.outer(np.ones(np.size(u)), np.cos(v))

# Generate Tangent Planes
grid = np.linspace(-1.2, 1.2, 10)
X, Y = np.meshgrid(grid, grid)
Z_max = (c_max - n_vec[0]*X - n_vec[1]*Y) / n_vec[2]
Z_min = (-c_max - n_vec[0]*X - n_vec[1]*Y) / n_vec[2]

# Plot Ellipsoid
ax.plot_surface(x_ell, y_ell, z_ell, color='cyan', alpha=0.2, linewidth=0)

# Plot Planes
ax.plot_surface(X, Y, Z_max, color='orange', alpha=0.4)
ax.plot_surface(X, Y, Z_min, color='blue', alpha=0.4)



\end{lstlisting}
